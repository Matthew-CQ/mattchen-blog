wget下载google文件 ​

bash
# wget下载google文件，参考https://lccurious.github.io/2021/05/15/Download-from-Google-Drive/
#!/bin/bash

# cd scratch place
# cd data/

# Download zip dataset from Google Drive
# 例如网址是https://drive.google.com/file/d/1bm6OOu-lE-NKyUFD5AaMZVLAeUbzI-_m/view?usp=sharing
# 文件为20130110.bag
filename='20130110.bag'
fileid='1bm6OOu-lE-NKyUFD5AaMZVLAeUbzI-_m'
wget -c --load-cookies /tmp/cookies.txt "<https://drive.google.com/uc?export=download&confirm=$>(wget --quiet --save-cookies /tmp/cookies.txt --keep-session-cookies --no-check-certificate '<https://drive.google.com/uc?export=download&id=${fileid}>' -O- | sed -rn 's/.confirm=([0-9A-Za-z_]+)./\\1\\n/p')&id=${fileid}" -O ${filename} && rm -rf /tmp/cookies.txt

# Unzip
# unzip -q ${filename}
# rm ${filename}
# cd

开发板挂载 ​

python
# 在板子上挂在ubuntu本地文件，避免拷贝，加快开发
# <https://blog.csdn.net/wojiushiwoba/article/details/70862342>
# 板子上挂在本地命令
sudo mount -t nfs -o nolock 172.16.0.88:/mnt/data/Work/ros2_ws/install /xdata/chenq/install
# 开机自动挂载
vim /etc/fstab
172.16.0.88:/mnt/data/Work/ros2_ws/src/perception/build/_install        /xdata/_install nfs     defaults,_netdev    0       0

# 本地跳板机挂载板子文目录
sshfs nvidia@172.16.0.26:/xdata/ /mnt/data/Work/orin/
# 让其他机子访问本机目录，完成目录跳转
sftp://pxing@10.68.68.226/mnt/data/Work/orin
